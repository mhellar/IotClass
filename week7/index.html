code/speak/app.js


//the child_process library let's us execute command line commands
var exec = require('child_process').exec;
//This variable stores the command we want to execute
//we are going to use the say command
var say = 'say ';

//let's make a function that speaks
function speak(whatosay){
    //speak the string
    exec(say + whatosay);
    //log it to the console
    console.log(whatosay)
}
// call the function
speak("I just do this");
